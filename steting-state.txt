ตัวอย่างเวลาเอาคลาสไปเรียกใช้ (เขียนโค้ดประมาณนี้)

Java
// สร้างกล่องเก็บข้อมูลขึ้นมา (หรือโหลดมาจากไฟล์เซฟ)
GameState currentState = new GameState();

// เพื่อนที่ทำระบบตั้งชื่อตัวละคร ก็จะเรียกใช้แบบนี้
currentState.playerName = "นายนัท";

// เพื่อนที่ทำระบบเนื้อเรื่อง (เพิ่มค่าความชอบ) ก็จะเรียกใช้แบบนี้
currentState.affection += 5;

// เพื่อนที่ทำระบบทางเลือก ก็จะบันทึกสถานะเหตุการณ์แบบนี้
currentState.setFlag("met_aoi", true);

// และเมื่อถึงจุดที่ต้องเช็คเงื่อนไข ก็จะดึงค่าไปเช็คแบบนี้
if (currentState.getFlag("met_aoi") == true) {
    System.out.println("Aoi: อ้าว เจอกันอีกแล้วนะ " + currentState.playerName);
} else {
    System.out.println("Aoi: สวัสดีค่ะ เพิ่งเคยเจอกันครั้งแรกใช่ไหม?");
}
